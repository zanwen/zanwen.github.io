停止模式-Stop

退出停止模式后，HSI会被选择作为系统时钟，因此我们需要重新打通HSE和PLL

示例代码

main.c

#include "delay.h"
#include "key.h"
#include "led.h"
#include "logger.h"
#include "stdio.h"
#include "stm32f10x.h"
#include "string.h"
#include "uart.h"

void uart1_received_callback(uint8_t buf[], uint8_t size) {}

void enter_stop_mode(void) {
    SCB->SCR |= SCB_SCR_SLEEPDEEP;

    RCC->APB1ENR |= RCC_APB1ENR_PWREN; // 配置PWR之前需要使能其时钟
    PWR->CR &= ~PWR_CR_PDDS;           // 停止模式
    PWR->CR |= PWR_CR_LPDS; // 停止模式下电压调节器为低功耗模式
    __WFI();                // 进入停止模式
}

void get_system_clock(uint32_t *sysclk, uint32_t *hclck, uint32_t *pclk1,
                      uint32_t *pclk2) {
    uint32_t rcc_config = RCC->CFGR;

    // 根据sysclk时钟源来计算其频率
    uint32_t sysclk_source = rcc_config & RCC_CFGR_SWS;
    if (sysclk_source == RCC_CFGR_SWS_HSI) {
        *sysclk = HSI_VALUE;
    } else if (sysclk_source == RCC_CFGR_SWS_HSE) {
        *sysclk = HSE_VALUE;
    } else if (sysclk_source == RCC_CFGR_SWS_PLL) {
        // HSE*PLL倍频系数
        uint8_t pllmul = ((rcc_config & RCC_CFGR_PLLMULL) >> 18) + 2;
        *sysclk = HSE_VALUE * pllmul;
    }
    // 根据AHP预分频计算hclk
    uint8_t hpre = (rcc_config & RCC_CFGR_HPRE) >> 4;
    if (hpre < 8) {
        *hclck = *sysclk;
    } else if (hpre < 12) {
        *hclck = *sysclk >> (hpre - 8 + 1);
    } else {
        *hclck = *sysclk >> (hpre - 8 + 2);
    }
    // 根据apb的预分频计算pclk
    uint8_t ppre1 = (rcc_config & RCC_CFGR_PPRE1) >> 8;
    uint8_t ppre2 = (rcc_config & RCC_CFGR_PPRE2) >> 11;
    if (ppre1 < 4) {
        *pclk1 = *sysclk;
    } else {
        *pclk1 = *sysclk >> (ppre1 - 4 + 1);
    }
    if (ppre2 < 4) {
        *pclk2 = *sysclk;
    } else {
        *pclk2 = *sysclk >> (ppre2 - 4 + 1);
    }
}

// 将系统时钟从HSI切换到HSE+PLL，参考SystemInit
void config_system_clock(void) {
    // 使能HSE
    RCC->CR |= RCC_CR_HSEON;
    // 等待HSE就绪
    while ((RCC->CR & RCC_CR_HSERDY) == 0) {
    }
    // 选择HSE作为PLL源
    RCC->CFGR |= RCC_CFGR_PLLSRC_HSE;
    // 关闭HSI
    RCC->CR &= ~RCC_CR_HSION;
    // 使能PLL
    RCC->CR |= RCC_CR_PLLON;
    // 等待PLL就绪
    while ((RCC->CR & RCC_CR_PLLRDY) == 0) {
    }
    // 选择PLL作为sysclk
    RCC->CFGR &= ~RCC_CFGR_SW;
    RCC->CFGR |= RCC_CFGR_SW_1;
    // 等待切换完成
    while ((RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_PLL) {
    }
}

int main() {
    uart_init();
    LED_Init();
    Key_Init();
    LOG_DEBUG("hello world\n")

    // 停止模式下，IO保持运行时状态
    LED_On(LED1);

    while (1) {
        delay_ms(3000); // 加个延时，给下次烧录预留点缓冲时间

        LOG_DEBUG("进入停止模式，可通过外部中断按键唤醒")
        enter_stop_mode();

        // 查看切换PLL前时钟频率
        uint32_t sysclk, hclck, pclk1, pclk2;
        get_system_clock(&sysclk, &hclck, &pclk1, &pclk2);

        config_system_clock(); // 切换HSI为PLL

        // 查看切换PLL后时钟频率
        uint32_t sysclk_after, hclck_after, pclk1_after, pclk2_after;
        get_system_clock(&sysclk_after, &hclck_after, &pclk1_after,
                         &pclk2_after);

        LOG_DEBUG("从停止模式退出")
        LOG_DEBUG("PLL切换前, sysclk = %d, hclk = %d, pclk1 = %d, pclk2 = %d",
                  sysclk, hclck, pclk1, pclk2)
        LOG_DEBUG("PLL切换后, sysclk = %d, hclk = %d, pclk1 = %d, pclk2 = %d",
                  sysclk_after, hclck_after, pclk1_after, pclk2_after)
    }
}

key.c

#include "key.h"
#include "delay.h"
#include "led.h"



void Key_Init(void) {
    // gpio clock
    RCC->APB2ENR |= RCC_APB2ENR_IOPFEN;
    // afio clock
    RCC->APB2ENR |= RCC_APB2ENR_AFIOEN;

    // gpio config for PF10
    GPIOF->CRH &= ~GPIO_CRH_MODE10;   // input mode
    GPIOF->CRH |= GPIO_CRH_CNF10_1;   // input with pull-up/down
    GPIOF->CRH &= ~GPIO_CRH_CNF10_0;
    GPIOF->ODR &= ~GPIO_ODR_ODR10;   // pull-down

    // afio config
    AFIO->EXTICR[2] &= ~AFIO_EXTICR3_EXTI10;
    AFIO->EXTICR[2] |= AFIO_EXTICR3_EXTI10_PF;   // select PF as exti10

    // exti config
    EXTI->IMR |= EXTI_IMR_MR10;   // unmask interrupt
    EXTI->RTSR |= EXTI_RTSR_TR10;   // rising edge trigger

    // nvic config
    NVIC_SetPriorityGrouping(3);   // only pre-priority
    NVIC_SetPriority(EXTI15_10_IRQn, 3);
    NVIC_EnableIRQ(EXTI15_10_IRQn);   // response interrupt
}

    }
}

待机模式-Standby

示例代码

void enter_standby_mode(void) {
    // 使能唤醒引脚(PA0)，引脚会被强制为输入下拉配置（最好提前使能GPIOA的时钟）
    PWR->CSR |= PWR_CSR_EWUP;

    SCB->SCR |= SCB_SCR_SLEEPDEEP;
    PWR->CR |= PWR_CR_PDDS;
    PWR->CR |= PWR_CR_CWUF; // CSR-WUF的清除需要通过置位CR-CWUF
    __WFI();
}

int main() {
    uart_init();
    LED_Init();
    // 需要访问PWR寄存器，提前开启时钟
    RCC->APB1ENR |= RCC_APB1ENR_PWREN;
    LOG_DEBUG("hello world")

    // SBF指示了曾经是否进入过待机模式
    if (PWR->CSR & PWR_CSR_SBF) {
        LOG_DEBUG("当前复位由待机模式下唤醒")
        PWR->CR |= PWR_CR_CSBF;
    } else {
        LOG_DEBUG("非待机唤醒复位")
    }

    // 待机模式下, IO为高阻态（除了复位、唤醒引脚）
    LED_On(LED2);

    while (1) {
        delay_ms(3000); // 加个延时，给下次烧录预留点缓冲时间
        LOG_DEBUG("进入待机模式，可通过复位和唤醒引脚退出")
        enter_standby_mode();
        LOG_DEBUG("退出待机模式") // 待机模式唤醒后，走复位流程
    }
}

总结

降低功耗方向：

  • 数字信号时钟
    • CPU时钟：停止执行指令
    • 外设时钟：外设停止工作
  • 高速晶振：唤醒后需要将HSI切换为HSE+PLL
  • 电压调节器
    • 寄存器、SRAM保护：决定唤醒后能否恢复现场继续执行

POR 和 PDR 的区别

特性 Power-on Reset (POR) Power-down Reset (PDR)
触发条件 上电时，电压从 0 上升到工作电压范围 电源电压下降到芯片最低工作电压以下
工作阶段 上电初始阶段 运行过程中
作用 确保芯片在电压足够高时才开始运行 保护芯片在欠压情况下停止运行，防止错误执行
典型应用场景 芯片启动时（比如接入电池或插上外部电源） 运行过程中发生电源波动或断电
复位时机 电压达到指定门限时释放复位信号 电压回升到指定门限以上时释放复位信号
复位阈值 通常比 PDR 的电压门限稍高 通常比 POR 的电压门限稍低

POR 和 PDR 的重要性

这两种复位机制在系统稳定性中扮演着重要角色：

Power-on Reset 的重要性

  • 确保芯片只有在供电电压达到安全范围后才会运行。
  • 避免芯片在电压未稳定时尝试执行指令，导致系统崩溃或初始化失败。

Power-down Reset 的重要性

  • 防止在电源电压过低时运行 MCU，避免损坏外设或执行错误代码。
  • 在电池供电系统中，保护用户数据或触发安全关机流程。
